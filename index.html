Supabase クラウド同期

設定すると複数デバイスでデータを共有できます。無料で利用できます。

1
Supabaseでプロジェクトを作成
supabase.com → New Project を作成してください。
2
SQL Editorで下のSQLを実行
右のSQLをコピーして Supabase SQL Editor に貼り付け実行してください。
3
URLとAnon Keyを入力
Settings → API からコピーして入力してください。

テーブル作成 SQL 

create table task_states (
  task_id text primary key,
  status text default 's-todo',
  progress text default '0%',
  deadline text default '',
  desc text default '',
  tool text default '',
  next_action text default '',
  updated_at timestamptz default now()
);

create table revenues (
  id bigint generated always as identity primary key,
  date text not null,
  desc text not null,
  cat text not null,
  amount numeric not null,
  type text not null,
  created_at timestamptz default now()
);

create table reports (
  id bigint generated always as identity primary key,
  title text not null,
  tag text default '',
  body text not null,
  created_at timestamptz default now()
);

alter table task_states disable row level security;
alter table revenues disable row level security;
alter table reports disable row level security;
